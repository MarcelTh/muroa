00042 CPlaylistItem::CPlaylistItem(CRootItem *root_item, std::string text, CCategoryItem*  parent, int posInParent) throw(ExMalformedPatch)
00043 : IContentItem( root_item, parent, CItemType::E_PLAYLISTITEM )
00044 {
00045 
00046         m_text = text;
00047         // first section is handled by CItemBase
00048         size_t lpos, rpos;
00049         // lpos = m_text.find('\t', 1) + 1;
00050         lpos = 0;
00051 
00052         try {
00053                 rpos = m_text.find('\t', lpos);
00054                 if( rpos != string::npos ) {
00055                         string mediaitemHashStr = text.substr(lpos, rpos - lpos);
00056                         m_mediaitem_hash = CUtils::str2uint32(mediaitemHashStr.c_str());
00057                         lpos = rpos + 1;
00058                 }
00059                 else {
00060                         throw ExMalformedPatch("error parsing 'media item hash' field, terminating tab char is missing." ,-1);
00061                 }
00062 
00063                 string hashStr;
00064                 rpos = m_text.find('\t', lpos);
00065                 if( rpos != string::npos ) {
00066                         hashStr = text.substr(lpos, rpos - lpos);
00067                         lpos = rpos + 1;
00068                 }
00069                 else {
00070                         hashStr = text.substr(lpos);
00071                 }
00072                 m_hash = CUtils::str2uint32(hashStr.c_str());
00073                 if(m_hash == 0){
00074                         m_hash = m_next_free_id++;
00075                 }
00076 
00077                 assembleText();
00078                 if(m_parent) {
00079                         m_parent->addChild(this, posInParent);
00080                 }
00081 
00082                 // for CPlaylistItem, m_hash is an index number that never changes during
00083                 // livetime of an object.
00084                 m_root_item->setContentPtr(CItemType(CItemType::E_PLAYLISTITEM), this, m_hash );
00085         }
00086         catch(std::invalid_argument& ex) {
00087                 throw ExMalformedPatch(ex.what(), -1);
00088         }
